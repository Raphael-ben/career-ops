%% Sidebar section title — more space above (separates from prev block), tight below
\newcommand{\sidebarSection}[1]{%
  \vspace{16pt}%
  {\centering\normalsize\textbf{#1}\par}%
  \vspace{5pt}%
}

%% Single source of truth for sidebar body text size — used by \sline, \pair, \address, \phone, \email
%% Change this once to re-size every sidebar body item (section titles use \normalsize and are unaffected).
\newcommand{\sidebarBodyFont}{\small}

%% Sidebar two-column row — used internally by \pair inside sidebarBlock
\newcommand{\skillrow}[2]{%
  \noindent\begin{tabular}{@{}>{\centering\arraybackslash}p{0.47\linewidth}@{\hspace{0.06\linewidth}}>{\centering\arraybackslash}p{0.47\linewidth}@{}}
    \sidebarBodyFont#1 & \sidebarBodyFont#2\\
  \end{tabular}\par\vspace{\sidebarLineGap}%
}

%% --- Sidebar building blocks ---
%% These hide all visual mechanics (centering, line gaps, vfill, photo crop,
%% mailto bug workaround). Body authors only call the semantic commands below.

%% \cvName — renders the name centred at LARGE, with bottom gap.
\newcommand{\cvName}{%
  {\centering\LARGE\textbf{\CVName}\par}%
  \vspace{\nameBelowGap}%
}

%% \cvPhoto — renders circular photo crop or placeholder fallback.
%%   Crop controlled by preamble: \photoRadius, \photoScale, \photoXshift, \photoYshift.
\newcommand{\cvPhoto}{%
  \begin{center}%
  \IfFileExists{\CVPhoto}{%
    \begin{tikzpicture}
      \useasboundingbox (-\photoRadius,-\photoRadius) rectangle (\photoRadius,\photoRadius);
      \clip (0,0) circle (\photoRadius);
      \node[inner sep=0] at (\photoXshift,\photoYshift)
        {\includegraphics[width=\photoScale]{\CVPhoto}};
    \end{tikzpicture}%
  }{%
    \begin{tikzpicture}
      \useasboundingbox (-\photoRadius,-\photoRadius) rectangle (\photoRadius,\photoRadius);
      \draw[color=SubtleGray] (0,0) circle (\photoRadius);
      \node[color=SubtleGray,font=\small] at (0,0) {[photo]};
    \end{tikzpicture}%
  }%
  \end{center}\vspace{\photoBelowGap}%
}

%% \tagline{text} — renders centred bold tagline with bottom gap before first block.
\newcommand{\tagline}[1]{%
  {\centering\large\textbf{#1}\par}%
  \vspace{\taglineBelowGap}%
}

%% sidebarBlock — header + centred body. Body uses \sline (single column) or \pair (two column).
%%   \begin{sidebarBlock}{TITLE}
%%     \sline{single line}
%%     \pair{left}{right}
%%   \end{sidebarBlock}
\newenvironment{sidebarBlock}[1]{%
  \sidebarSection{#1}%
  \centering
}{%
  \par
}

%% \sline{text} — single-column item inside sidebarBlock or contactBlock.
\newcommand{\sline}[1]{{\sidebarBodyFont #1}\par\vspace{\sidebarLineGap}}

%% \pair{left}{right} — two-column item inside sidebarBlock (alias for \skillrow).
\let\pair\skillrow

%% contactBlock — automatically pushed to bottom of sidebar via \vfill.
%%   \begin{contactBlock}
%%     \address{...}  \phone{...}  \email{...}
%%   \end{contactBlock}
\newenvironment{contactBlock}{%
  \sidebarSection{CONTACT DETAILS}%
  \centering
}{%
  \par\vspace{\sidebarBottomPad}%
}

%% \address{text} — postal address line. Use \\ for line breaks inside.
\newcommand{\address}[1]{{\sidebarBodyFont #1}\par\vspace{2pt}}

%% \phone{number} — phone with leading icon. Plain text (NOT a tel: link).
%%   Mac Preview expands hyperref link annotations to the full container minipage,
%%   making clickable links unreliable in the sidebar. Icon is purely visual.
\newcommand{\phone}[1]{{\sidebarBodyFont\faPhone~#1}\par}

%% \email{addr} — email with mailto: link. Mac Preview tends to expand the
%%   hyperref annotation rectangle in centered minipage contexts. Two-pronged
%%   defense: (a) \mbox{} forces a single tight inline box around the content,
%%   (b) pdflinkmargin=0pt in \hypersetup zeroes hyperref's annotation padding.
%%   If Preview still misbehaves, revert to plain text: {\sidebarBodyFont\faEnvelope~#1}.
\newcommand{\email}[1]{{\sidebarBodyFont\href{mailto:#1}{\mbox{\faEnvelope~#1}}}\par}

%% Main section title with optional FontAwesome icon
%%   \mainSection[\faBriefcase]{TITLE}   — with icon
%%   \mainSection{TITLE}                 — no icon
\newcommand{\mainSection}[2][\relax]{%
  \noindent\Large\textbf{%
    \ifx#1\relax\else{\normalsize\color{black}#1\hspace{5pt}}\fi%
    #2%
  }\par\vspace{\sectionRuleGap}%
  \noindent\rule{\linewidth}{0.5pt}\par\vspace{\sectionBodyGap}%
}

%% Experience / Education block — atomic unit: title row + company row + bullets
%%   \begin{experience}{date}{title}{org}{location}{url}
%%     \item bullet
%%     \item bullet
%%   \end{experience}
%% Use \begin{education}{...} for education entries (semantic alias of experience).
%% All internal spacing (\entryRowGap, \entryBottomGap, itemsep, \entryGap) is
%% controlled by named variables in the Spacing controls block above.
\newenvironment{experience}[5]{%
  \noindent\begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
    {\large\textbf{#2}} & {\small\color{SubtleGray}#1}%
  \end{tabular*}\par\vspace{\entryRowGap}%
  \noindent\begin{tabular*}{\linewidth}{@{}l@{\extracolsep{\fill}}r@{}}
    {\small\href{#5}{\color{CompanyColor}\ul{#3}}} & {\small\color{LocationColor}#4}%
  \end{tabular*}\par\vspace{\entryBottomGap}%
  \begin{itemize}[%
    leftmargin=1.2em,%
    label={\small$\cdot$},%
    itemsep=2pt,%
    parsep=0pt,%
    topsep=0pt,%
    partopsep=0pt%
  ]\small%
}{%
  \end{itemize}\vspace{\entryGap}%
}
